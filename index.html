Ejemplos de Sintaxis en JavaScript

Control de Flujo y Manejo de Errores

if...else


if (true) {
    console.log("Esto es verdadero");
} else {
    console.log("Esto es falso");
}

switch


let fruit = "apple";
switch (fruit) {
    case "banana":
        console.log("Banana");
        break;
    case "apple":
        console.log("Apple");
        break;
    default:
        console.log("Other fruit");
}

try/catch/throw


try {
    throw new Error("Algo salió mal");
} catch (error) {
    console.error(error.message);
}

Bucles e Iteración

for


for (let i = 0; i < 5; i++) {
    console.log(i);
}

while


let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

do...while


let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

Funciones

Definir y Llamar Funciones


function greet(name) {
    return `Hello, ${name}!`;
}
console.log(greet("World"));

Expresiones y Operadores

Asignación y Comparación


let x = 5;
let y = 10;
console.log(x == y); // false
console.log(x != y); // true

Operadores Aritméticos


let sum = x + y;
let diff = x - y;
console.log(sum); // 15
console.log(diff); // -5

Operadores Lógicos


console.log(true && false); // false
console.log(true || false); // true
console.log(!true); // false

Números y Fechas

Objeto Number


let num = Number("123");
console.log(num); // 123

Objeto Math


console.log(Math.PI); // 3.141592653589793
console.log(Math.sqrt(16)); // 4

Objeto Date


let now = new Date();
console.log(now.toString());

Formateo de Texto

Objeto String


let str = "Hola, Mundo";
console.log(str.toUpperCase()); // HOLA, MUNDO
console.log(str.toLowerCase()); // hola, mundo

Colecciones Indexadas

Arreglos


let arr = [1, 2, 3, 4, 5];
console.log(arr.length); // 5
console.log(arr[0]); // 1

DOM

Encontrar elementos HTML


// Por ID
let element = document.getElementById("myId");
console.log(element);

// Por nombre de etiqueta
let elements = document.getElementsByTagName("p");
console.log(elements);

// Por nombre de clase
elements = document.getElementsByClassName("myClass");
console.log(elements);

// Por selectores CSS
element = document.querySelector(".myClass");
console.log(element);

// Por colecciones de objetos HTML
elements = document.querySelectorAll(".myClass");
console.log(elements);
